// CORS Proxy Services (rotate between them for reliability) const CORS_PROXIES = [ 'https://api.allorigins.win/raw?url=', 'https://corsproxy.io/?', 'https://cors-anywhere.herokuapp.com/' ]; // Real RSS Feed Fetcher class RSSFeedManager { constructor() { this.currentProxyIndex = 0; } else { console.warn(`❌ Failed to fetch ${batch[index]}:`, result.reason); } }); } catch (error) { console.error('Batch processing error:', error); } // Update progress const progress = Math.min(((i + batchSize) / allFeeds.length) * 100, 100); this.updateProgress(`Fetching feeds... ${Math.round(progress)}%`); // Small delay between batches to avoid overwhelming servers if (i + batchSize < allFeeds.length) { await this.delay(200); } } console.log(`🎉 RSS feed collection complete! Found ${stories.length} total items`); return this.filterCorruptionStories(stories); } async fetchSingleFeed(feedUrl) { // Check cache first const cached = this.getCachedFeed(feedUrl); if (cached) { return cached; } for (let attempt = 0; attempt < this.maxRetries; attempt++) { try { const proxyUrl = this.getNextProxy() + encodeURIComponent(feedUrl); const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), 10000); // 10s timeout const response = await fetch(proxyUrl, { signal: controller.signal, headers: { 'User-Agent': 'Mozilla/5.0 (compatible; CorruptionTracker/1.0)' } }); clearTimeout(timeoutId); if (!response.ok) { throw new Error(`HTTP ${response.status}: ${response.statusText}`); } const xmlText = await response.text(); const stories = this.parseRSSXML(xmlText, feedUrl); // Cache successful results this.cacheData(feedUrl, stories); return stories; } catch (error) { console.warn(`Attempt ${attempt + 1} failed for ${feedUrl}:`, error.message); if (attempt === this.maxRetries - 1) { throw error; } await this.delay(1000 * (attempt + 1)); // Exponential backoff } } } parseRSSXML(xmlText, sourceUrl) { try { const parser = new DOMParser(); const xmlDoc = parser.parseFromString(xmlText, 'text/xml'); // Handle parsing errors const parserError = xmlDoc.querySelector('parsererror'); if (parserError) { throw new Error('XML parsing failed: ' + parserError.textContent); } const items = xmlDoc.querySelectorAll('item, entry'); const stories = []; items.forEach(item => { try { const title = this.getTextContent(item, 'title'); const link = this.getTextContent(item, 'link, guid'); const description = this.getTextContent(item, 'description, summary, content'); const entities = this.extractEntities(title + ' ' + cleanDesc); stories.push({ id: this.generateId(link, pubDate), headline: title.trim(), url: link.trim(), summary: cleanDesc.substring(0, 500), date: this.parseDate(pubDate), state: this.extractState(fullText), type: 'news', subtype: this.classifyCorruptionType(fullText), amount: this.extractAmount(fullText), parties: entities.persons, flags: this.identifyRedFlags(fullText), severity: this.assessSeverity(fullText, entities.amounts), entities: entities, source: this.identifySource(sourceUrl), riskScore: this.calculateRiskScore(fullText, entities) }); for (const [type, keywords] of Object.entries(patterns)) { if (keywords.some(keyword => text.includes(keyword))) { return type; } } return 'general'; } extractEntities(text) { const entities = { persons: [], organizations: [], amounts: [] }; // Extract dollar amounts const amountRegex = /\$[\d,]+(?:\.\d{2})?(?:\s*(?:million|billion|thousand))?/gi; const amountMatches = text.match(amountRegex) || []; entities.amounts = amountMatches.map(match => { let value = parseFloat(match.replace(/[\$,]/g, '')); const lowerMatch = match.toLowerCase(); if (lowerMatch.includes('million')) value *= 1000000; if (lowerMatch.includes('billion')) value *= 1000000000; if (lowerMatch.includes('thousand')) value *= 1000; return value; }); // Extract officials and titles const titleRegex = /(Mayor|Governor|Sheriff|Chief|Commissioner|Representative|Senator|Judge|Councilman|Superintendent|Director|Treasurer|Clerk)\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)?)/g; let match; while ((match = titleRegex.exec(text)) !== null) { entities.persons.push(match[1] + ' ' + match[2]); } // Extract organizations const orgRegex = /([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(?:Corporation|Company|LLC|Inc|Department|Agency|Bureau|Commission)/g; while ((match = orgRegex.exec(text)) !== null) { entities.organizations.push(match[0]); } return entities; } extractState(text) { const statePatterns = { 'alabama': 'AL', 'alaska': 'AK', 'arizona': 'AZ', 'arkansas': 'AR', 'california': 'CA', 'colorado': 'CO', 'connecticut': 'CT', 'delaware': 'DE', 'florida': 'FL', 'georgia': 'GA', 'hawaii': 'HI', 'idaho': 'ID', 'illinois': 'IL', 'indiana': 'IN', 'iowa': 'IA', 'kansas': 'KS', 'kentucky': 'KY', 'louisiana': 'LA', 'maine': 'ME', 'maryland': 'MD', 'massachusetts': 'MA', 'michigan': 'MI', 'minnesota': 'MN', 'mississippi': 'MS', 'missouri': 'MO', 'montana': 'MT', 'nebraska': 'NE', 'nevada': 'NV', 'new hampshire': 'NH', 'new jersey': 'NJ', 'new mexico': 'NM', 'new york': 'NY', 'north carolina': 'NC', 'north dakota': 'ND', 'ohio': 'OH', 'oklahoma': 'OK', 'oregon': 'OR', 'pennsylvania': 'PA', 'rhode island': 'RI', 'south carolina': 'SC', 'south dakota': 'SD', 'tennessee': 'TN', 'texas': 'TX', 'utah': 'UT', 'vermont': 'VT', 'virginia': 'VA', 'washington': 'WA', 'west virginia': 'WV', 'wisconsin': 'WI', 'wyoming': 'WY' }; if (matches) { for (const match of matches) { if (Object.values(statePatterns).includes(match)) { return match; } } } return null; } // Helper methods getTextContent(element, selectors) { const selectorArray = selectors.split(',').map(s => s.trim()); for (const selector of selectorArray) { const found = element.querySelector(selector); if (found) { return found.textContent || found.getAttribute('href') || ''; } } return ''; } cleanHtmlContent(html) { const div = document.createElement('div'); div.innerHTML = html; return div.textContent || div.innerText || ''; } parseDate(dateString) { if (!dateString) return new Date().toISOString(); try { return new Date(dateString).toISOString(); } catch (e) { return new Date().toISOString(); } } extractAmount(text) { const amounts = this.extractEntities(text).amounts; return amounts.length > 0 ? Math.max(...amounts) : null; } identifyRedFlags(text) { const flags = []; if (text.includes('single bid') || text.includes('sole bidder')) flags.push('singleBid'); if (text.includes('family') || text.includes('relative')) flags.push('familyTies'); if (text.includes('cash') || text.includes('unreported')) flags.push('unreported'); if (text.includes('federal') || text.includes('fbi')) flags.push('federalInvestigation'); if (text.includes('offshore') || text.includes('shell company')) flags.push('offshore'); if (text.includes('emergency') || text.includes('no-bid')) flags.push('emergencyContract'); return flags; } assessSeverity(text, amounts) { let score = 0; if (text.includes('federal') || text.includes('indictment')) score += 30; if (text.includes('fbi') || text.includes('grand jury')) score += 25; if (amounts.length > 0 && Math.max(...amounts) > 1000000) score += 20; if (text.includes('conspiracy') || text.includes('racketeering')) score += 15; if (text.includes('investigation') || text.includes('probe')) score += 10; if (score >= 25) return 'HIGH'; if (score >= 10) return 'MEDIUM'; return 'LOW'; } calculateRiskScore(text, entities) { let score = 0; if (text.includes('federal')) score += 20; if (text.includes('indictment') || text.includes('arrest')) score += 25; if (entities.amounts.length > 0) { const maxAmount = Math.max(...entities.amounts); if (maxAmount > 10000000) score += 30; else if (maxAmount > 1000000) score += 20; else if (maxAmount > 100000) score += 10; } score += entities.persons.length * 5; score += entities.organizations.length * 3; return Math.min(score, 100); } identifySource(url) { if (url.includes('justice.gov')) return 'DOJ'; if (url.includes('fbi.gov')) return 'FBI'; if (url.includes('sec.gov')) return 'SEC'; if (url.includes('abcnews.com')) return 'ABC News'; if (url.includes('cbsnews.com')) return 'CBS News'; if (url.includes('nytimes.com')) return 'NY Times'; if (url.includes('reuters.com')) return 'Reuters'; if (url.includes('citizensforethics.org')) return 'CREW'; if (url.includes('whistleblower.org')) return 'Whistleblower.org'; if (url.includes('spotlightpa.org')) return 'Spotlight PA'; return 'News Source'; } generateId(url, date) { return btoa((url + date + Math.random())).replace(/[^a-zA-Z0-9]/g, '').substring(0, 16);
